sftp> mkdir incident_data        # Create remote directory
sftp> rmdir old_logs            # Remove remote directory
sftp> exit                       # Close session

🚨 Emergency Collection Script:
#!/bin/bash
# Emergency Incident Response Collection
INCIDENT_ID="INC-$(date +%Y%m%d-%H%M%S)"
COLLECTION_DIR="/tmp/$INCIDENT_ID"

echo "EMERGENCY COLLECTION STARTED: $INCIDENT_ID"
mkdir -p "$COLLECTION_DIR"

# Critical system state
ps aux > "$COLLECTION_DIR/running_processes.txt"
netstat -tuln > "$COLLECTION_DIR/network_connections.txt"
mount > "$COLLECTION_DIR/mounted_filesystems.txt"

# Archive critical data
tar -czf "$COLLECTION_DIR/logs.tar.gz" /var/log/ 2>/dev/null
tar -czf "$COLLECTION_DIR/configs.tar.gz" /etc/ 2>/dev/null
tar -czf "$COLLECTION_DIR/home_dirs.tar.gz" /home/ 2>/dev/null

# Generate integrity hashes
sha256sum "$COLLECTION_DIR"/*.tar.gz > "$COLLECTION_DIR/checksums.txt"

echo "COLLECTION COMPLETE: $COLLECTION_DIR"
echo "Transfer immediately to secure location!"

📅 Common Compliance Retention Requirements:
SOX (Sarbanes-Oxley): 7 years for financial audit logs
HIPAA: 6 years for healthcare access logs
PCI-DSS: 1 year for payment card data logs
GDPR: Varies by purpose, deletion required when no longer needed
SOC 2: Varies by control, typically 1-3 years
📊 Compliance Best Practices
Automation: Use cron jobs for regular archival
Integrity: Always generate and verify cryptographic hashes
Immutability: Store archives on write-once media when possible
Access Controls: Restrict archive access to authorized personnel only
