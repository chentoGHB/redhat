# DISA STIG Compliant RHEL 9 Installation # Security-First Configuration # FIPS Mode for Cryptographic Compliance fips --enable # SELinux Mandatory Access Controls selinux --enforcing # Firewall with Strict Rules firewall --enabled --ssh --remove-service=dhcpv6-client # Encrypted Storage with Strong Algorithms zerombr clearpart --all --initlabel autopart --type=lvm --encrypted --passphrase=STRONG_PASSPHRASE # Strong Authentication auth --enableshadow --passalgo=sha512 --enablefips # Account Security Policies %addon org.fedoraproject.Anaconda.addon.pwpolicy pwpolicy root --minlen=12 --minquality=50 --strict --nochanges --notempty pwpolicy user --minlen=12 --minquality=50 --strict --nochanges --emptyok %end # Security-Focused Package Selection %packages @core @security-tools aide openscap-scanner scap-security-guide audispd-plugins rsyslog-gnutls chrony -telnet-server -rsh-server -ypserv %end # Post-Install Security Hardening %post --log=/root/security-hardening.log # Install SCAP security profiles dnf install -y scap-security-guide # Enable security services systemctl enable aide systemctl enable auditd systemctl enable rsyslog # Apply DISA STIG Profile oscap xccdf eval --profile=stig \ --results=/tmp/stig-results.xml \ /usr/share/xml/scap/ssg/content/ssg-rhel9-ds.xml echo "STIG-compliant system deployed: $(date)" > /etc/motd %end

🏢 Enterprise Deployment Monitoring:
#!/bin/bash # Enterprise Security Infrastructure Deployment Monitor echo "=== Global Security Deployment Status ===" echo "Timestamp: $(date -u '+%Y-%m-%d %H:%M:%S UTC')" echo # Regional Deployment Status for region in americas emea apac; do echo "=== Region: $region ===" # Kickstart Server Status kickstart_server="kickstart-$region.corp.com" if curl -s --connect-timeout 5 "http://$kickstart_server/health" > /dev/null; then echo "✓ Kickstart Server: ONLINE" active_deployments=$(curl -s "http://$kickstart_server/api/deployments" | jq '.active_count') echo " Active Deployments: $active_deployments" else echo "✗ Kickstart Server: OFFLINE" fi # Security Systems Count security_systems=$(ansible-inventory --list | jq ".security_systems_$region.hosts | length") echo " Security Systems: $security_systems" # Compliance Status compliance_rate=$(oscap-report --region $region --compliance-rate) echo " STIG Compliance: $compliance_rate%" echo done # Global Statistics total_systems=$(ansible-inventory --list | jq '[.security_systems_americas.hosts, .security_systems_emea.hosts, .security_systems_apac.hosts] | flatten | length') echo "=== Global Summary ===" echo "Total Security Systems: $total_systems" echo "Average Deployment Time: $(deployment-metrics --average-time)" echo "Global Compliance Rate: $(oscap-report --global --compliance-rate)%"
