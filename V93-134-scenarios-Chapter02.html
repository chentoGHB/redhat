🛡️ Security Baseline Script:
#!/bin/bash
# Daily Security Baseline Check

# Check for unauthorized SUID files
find / -type f -perm -4000 2>/dev/null > /var/log/security/suid_files.log

# Verify critical service status
for service in firewalld sshd; do
    if ! systemctl is-active --quiet $service; then
        logger "SECURITY ALERT: $service is not running"
        echo "Service $service is down!" | mail security@company.com
    fi
done

# Check for failed login attempts in last 24 hours
grep "$(date --date='1 day ago' +%b\ %d)" /var/log/auth.log | \
grep "Failed password" > /var/log/security/daily_failed_logins.log

# Generate compliance report
echo "Security baseline check completed: $(date)" >> /var/log/security/compliance.log

🚨 Brute Force Detection Script:
#!/bin/bash
# Detect brute force login attempts

THRESHOLD=10
LOGFILE="/var/log/auth.log"
ALERT_FILE="/var/log/security/brute_force_alerts.log"

# Find IPs with more than THRESHOLD failed attempts in last 5 minutes
grep "$(date --date='5 minutes ago' +%b\ %d\ %H:%M)" "$LOGFILE" | \
grep "Failed password" | \
awk '{print $11}' | sort | uniq -c | \
while read count ip; do
    if [ "$count" -gt "$THRESHOLD" ]; then
        echo "$(date): BRUTE FORCE DETECTED from $ip ($count attempts)" >> "$ALERT_FILE"
        # Auto-block the IP (optional)
        # sudo iptables -A INPUT -s "$ip" -j DROP
        logger "SECURITY ALERT: Brute force attack from $ip ($count attempts)"
    fi
done
