🏢 Enterprise Health Monitoring Script:
#!/bin/bash # Security Container Health Monitoring echo "=== Security Container Health Check ===" echo "Timestamp: $(date)" echo # Check Security Services for service in container-security-collector container-siem-agent container-threat-intel; do echo "=== Service: $service ===" # Service Status if systemctl --user is-active $service >/dev/null 2>&1; then echo "✓ Status: ACTIVE" else echo "✗ Status: INACTIVE" fi # Service Enabled if systemctl --user is-enabled $service >/dev/null 2>&1; then echo "✓ Enabled: YES" else echo "✗ Enabled: NO" fi # Container Health container_name=$(echo $service | sed 's/container-//') if podman healthcheck run $container_name >/dev/null 2>&1; then echo "✓ Health: HEALTHY" else echo "⚠ Health: CHECK REQUIRED" fi echo done # Resource Usage echo "=== Resource Usage ===" podman stats --no-stream --format "table {{.Name}}\t{{.CPUPerc}}\t{{.MemUsage}}"
